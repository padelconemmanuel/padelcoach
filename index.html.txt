  return `${M[+m-1]} ${y}`;
}
function fmtMoney(n) { return '$' + Number(n).toLocaleString('es-AR'); }
function toHoy(s) {
  const hoy = new Date(); hoy.setHours(0,0,0,0);
  const d = new Date(s + 'T12:00:00');
  const diff = Math.round((d-hoy)/86400000);
  if (diff===0) return 'hoy'; if (diff===1) return 'mañana'; if (diff===-1) return 'ayer';
  return diff<0 ? `hace ${-diff} días` : `en ${diff} días`;
}

function rClases(selDate) {
  const sel = selDate || fmt(new Date());
  const d   = new Date(sel+'T12:00:00');
  S.calDate = d;
 
  // Calendar grid
  const y = d.getFullYear(), m = d.getMonth();
  const first = new Date(y,m,1).getDay();
  const last  = new Date(y,m+1,0).getDate();
  const prev  = new Date(y,m,0).getDate();
  const hoy   = fmt(new Date());
  const meses = ['Enero','Febrero','Marzo','Abril','Mayo','Junio','Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre'];
  const DOWS  = ['Dom','Lun','Mar','Mié','Jue','Vie','Sáb'];
 
  let cells = '';
  DOWS.forEach(d => { cells += `<div class="cal-dow">${d}</div>`; });
  for (let i=0; i<first; i++) {
    const dN = prev-first+1+i;
    const dateStr = fmt(new Date(y,m-1,dN));
    const has = S.clases.some(c=>c.fecha===dateStr);
    cells += `<div class="cal-day other${has?' has':''}" onclick="rClases('${dateStr}')">${dN}</div>`;
  }
  for (let i=1; i<=last; i++) {
    const dateStr = fmt(new Date(y,m,i));
    const has = S.clases.some(c=>c.fecha===dateStr);
    const isSel = dateStr===sel;
    const isTod = dateStr===hoy;
    const isPast = dateStr<hoy;
    cells += `<div class="cal-day${isSel?' sel':''}${isTod&&!isSel?' today':''}${has?' has':''}${isPast&&!isSel?' past':''}" onclick="rClases('${dateStr}')">${i}</div>`;
  }
  const remaining = 42 - first - last;
  for (let i=1; i<=remaining; i++) {
    const dateStr = fmt(new Date(y,m+1,i));
    const has = S.clases.some(c=>c.fecha===dateStr);
    cells += `<div class="cal-day other${has?' has':''}" onclick="rClases('${dateStr}')">${i}</div>`;
  }
 
  // Clases del día seleccionado
  const clasesDia = S.clases.filter(c=>c.fecha===sel).sort((a,b)=>a.hora.localeCompare(b.hora));
  const claseRows = clasesDia.map(cl => {
    const al = cl.alumnoId ? S.alumnos.find(a=>a.id===cl.alumnoId) : null;
    return `<div class="row" onclick="verClase(${cl.id})">
      <div class="row-time">${cl.hora}<br><span style="font-size:10px">${cl.dur}min</span></div>
      <div class="row-info"><div class="row-name">${cl.titulo}</div><div class="row-sub">${cl.cancha}${al?' · '+al.nombre.split(' ')[0]:' · grupal'}</div></div>
      ${BADGE[cl.estado]||''}
    </div>`;
  }).join('') || '<p class="empty">Sin clases este día</p>';
 
  const isSelHoy = sel===hoy;
  const labelDia = isSelHoy ? 'Hoy' : fmtDisp(sel);
 
  document.getElementById('sec-clases').innerHTML = `
    <div class="card">
      <div class="cal-month">
        <button class="cal-nav" onclick="rClases('${fmt(new Date(y,m-1,1))}')">‹</button>
        <h3>${meses[m]} ${y}</h3>
        <button class="cal-nav" onclick="rClases('${fmt(new Date(y,m+1,1))}')">›</button>
      </div>
      <div class="cal-grid">${cells}</div>
    </div>
    <div class="t-sec">${labelDia} — ${clasesDia.length} clase${clasesDia.length!==1?'s':''}</div>
    <div class="card-flush">${claseRows}</div>
    <button class="btn-outline" onclick="formClase('${sel}')">+ Agregar clase el ${labelDia}</button>
    <button class="btn-gcal" onclick="openGcal()">
      <svg viewBox="0 0 24 24" width="18" height="18" fill="none" stroke="currentColor" stroke-width="2"><rect x="3" y="4" width="18" height="18" rx="2"/><line x1="16" y1="2" x2="16" y2="6"/><line x1="8" y1="2" x2="8" y2="6"/><line x1="3" y1="10" x2="21" y2="10"/></svg>
      Sincronizar con Google Calendar

// Guardar registro de no-pago por clase
function registrarNoPago(alumnoId, claseId, claseTitle, fecha) {
  if (!window._clasesNoPago) window._clasesNoPago = JSON.parse(localStorage.getItem('clases_nopago') || '[]');
  const ya = window._clasesNoPago.find(r => r.alumnoId === alumnoId && r.claseId === claseId);
  if (!ya) {
    window._clasesNoPago.push({ alumnoId, claseId, claseTitle, fecha, ts: Date.now() });
    localStorage.setItem('clases_nopago', JSON.stringify(window._clasesNoPago));
  }
}
 
function saldarDeuda(alumnoId, claseId) {
  if (!window._clasesNoPago) window._clasesNoPago = JSON.parse(localStorage.getItem('clases_nopago') || '[]');

  localStorage.setItem('clases_nopago', JSON.stringify(window._clasesNoPago));

  toast('Deuda saldada ✓');
}

          <button class="pay-saldar-btn" onclick="abrirSaldarDeuda(${al.id},${d.claseId},'${d.claseTitle}')">Cobrar</button>
        </div>`
      ).join('');
 
      const histRows = pagosAl.map(p =>
        `<div class="pay-hist-row">
          <span>${fmtDisp(p.fecha)}</span>
          <span class="pay-metodo ${p.metodo==='Efectivo'?'ef':'tr'}">${p.metodo==='Efectivo'?'💵 Efectivo':'📲 Transf.'}</span>
          <span class="pay-monto-sm">${p.monto>0?fmtMoney(p.monto):'-'}</span>
        </div>`
      ).join('');
 
      return `
        <div class="pay-alumno-card ${tieneDeuda?'con-deuda':''}">
          <div class="pay-al-header" onclick="this.parentElement.classList.toggle('open')">
            ${avatarHtml(al,'sm')}
            <div class="pay-al-info">
              <div class="pay-al-name">${al.nombre}</div>
              <div class="pay-al-sub">${tieneDeuda?`<span style="color:var(--rd)">⚠️ ${deudas.length} clase${deudas.length>1?'s':''} sin pagar</span>`:(ultimoPago?`Último pago: ${fmtDisp(ultimoPago.fecha)}`:'Sin pagos registrados')}</div>
            </div>
            <div class="pay-al-estado ${tieneDeuda?'deuda':'ok'}">${tieneDeuda?'DEBE':'AL DÍA'}</div>
          </div>
          <div class="pay-al-body">
            ${tieneDeuda?`<div class="pay-deuda-section"><div class="pay-sec-lbl">Clases sin pagar</div>${deudaRows}</div>`:''}
            ${histRows?`<div class="pay-hist-section"><div class="pay-sec-lbl">Últimos pagos</div>${histRows}</div>`:''}
            <button class="pay-reg-btn" onclick="formPago(${al.id})">+ Registrar pago</button>
          </div>
        </div>`;
    }).join('') || '<p class="empty">Sin alumnos</p>';
 
    contenido = `
      ${alumnosDeuda.length ? `<div class="pay-alert-banner">⚠️ ${alumnosDeuda.length} alumno${alumnosDeuda.length>1?'s':''} con clases sin pagar</div>` : ''}
      <div class="pay-alumnos-list">${filas}</div>`;
 
  } else if (vista === 'movimientos') {
    const sorted = [...S.pagos].sort((a,b) => b.fecha.localeCompare(a.fecha));
    const rows = sorted.map(p => {
      const al = S.alumnos.find(a=>a.id===p.alumnoId);
      if (!al) return '';
      return `<div class="row" onclick="editarPago(${p.id})">
        ${avatarHtml(al,'sm')}
        <div class="row-info">
          <div class="row-name">${al.nombre}</div>
          <div class="row-sub">${fmtDisp(p.fecha)}${p.metodo?' · '+p.metodo:''}</div>
        </div>
        ${BADGE[p.estado]||''}
        <span class="row-amt" style="color:${p.estado==='pagado'?'var(--g)':'var(--am)'}">${p.monto>0?fmtMoney(p.monto):'-'}</span>
      </div>`;
    }).join('') || '<p class="empty">Sin movimientos</p>';
    contenido = `<div class="card-flush">${rows}</div>`;
 
  } else if (vista === 'resumen') {
    // Resumen mensual: efectivo vs transferencia + por alumno
    const meses = [];
    const mesSet = new Set(S.pagos.map(p=>p.fecha.slice(0,7)));
    mesSet.forEach(m => meses.push(m));
    meses.sort().reverse();
 
    const mesBars = meses.slice(0,4).map(m => {
      const ef = S.pagos.filter(p=>p.estado==='pagado'&&p.fecha.startsWith(m)&&p.metodo==='Efectivo').reduce((s,p)=>s+p.monto,0);
      const tr = S.pagos.filter(p=>p.estado==='pagado'&&p.fecha.startsWith(m)&&p.metodo==='Transferencia').reduce((s,p)=>s+p.monto,0);
      const tot = ef+tr||1;
      const efPct = Math.round(ef/tot*100);
      const trPct = 100-efPct;
      const [y,mo] = m.split('-');
      const nomMes = ['','Ene','Feb','Mar','Abr','May','Jun','Jul','Ago','Sep','Oct','Nov','Dic'][+mo];
      return `<div class="pay-mes-row">
        <div class="pay-mes-lbl">${nomMes} ${y}</div>
        <div class="pay-mes-bar-wrap">
          <div class="pay-mes-bar-ef" style="width:${efPct}%"></div>
          <div class="pay-mes-bar-tr" style="width:${trPct}%"></div>
        </div>
        <div class="pay-mes-tots"><span class="pay-metodo ef">💵 ${fmtMoney(ef)}</span><span class="pay-metodo tr">📲 ${fmtMoney(tr)}</span></div>
      </div>`;
    }).join('') || '<p class="empty">Sin datos</p>';
 
    // Ranking alumnos que más pagaron
    const ranking = S.alumnos.map(al => ({
      al,
      total: S.pagos.filter(p=>p.estado==='pagado'&&p.alumnoId===al.id).reduce((s,p)=>s+p.monto,0)
    })).filter(x=>x.total>0).sort((a,b)=>b.total-a.total);
 
    const rankRows = ranking.map(({al,total}) =>
      `<div class="row">${avatarHtml(al,'sm')}
        <div class="row-info"><div class="row-name">${al.nombre}</div></div>
        <span class="row-amt" style="color:var(--g)">${fmtMoney(total)}</span>
      </div>`
    ).join('') || '<p class="empty">Sin datos</p>';
 
    contenido = `
      <div class="metrics">
        <div class="met"><div class="met-lbl">Este mes</div><div class="met-val">${fmtMoney(pagadoMes)}</div><div class="met-sub">total</div></div>
        <div class="met"><div class="met-lbl">💵 Efectivo</div><div class="met-val">${fmtMoney(efectivoMes)}</div><div class="met-sub">este mes</div></div>
        <div class="met"><div class="met-lbl">📲 Transf.</div><div class="met-val">${fmtMoney(transfMes)}</div><div class="met-sub">este mes</div></div>
        <div class="met"><div class="met-lbl">📱 MP</div><div class="met-val">${fmtMoney(mpMes)}</div><div class="met-sub">este mes</div></div>
        <div class="met"><div class="met-lbl">📦 Paq/Cuota</div><div class="met-val">${fmtMoney(paqMes)}</div><div class="met-sub">este mes</div></div>
        <div class="met"><div class="met-lbl">Total histórico</div><div class="met-val">${fmtMoney(pagadoTotal)}</div><div class="met-sub">acumulado</div></div>
      </div>
      <div class="t-sec">Efectivo vs Transferencia por mes</div>
      <div class="card" style="padding:16px">${mesBars}</div>
      <div class="t-sec">Pagos por alumno (total)</div>
      <div class="card-flush">${rankRows}</div>`;
  }
 
  document.getElementById('sec-pagos').innerHTML = `
    <div class="pay-tabs">${tabsHtml}</div>
    ${contenido}
  `;
  document.getElementById('fab').onclick = () => formPago();
}

/* ═══════════════════════════════════════════════════════════════════
   NOTIFS
═══════════════════════════════════════════════════════════════════ */
function rNotifs() {
  const hoy = fmt(new Date());
  const man  = fmt(new Date(Date.now()+86400000));
  const notifs = [];
 
  S.pagos.filter(p=>p.estado==='vencido').forEach(p => {
    const al = S.alumnos.find(a=>a.id===p.alumnoId);
    if (al) notifs.push({ t:'r', msg:`Pago vencido — <strong>${al.nombre}</strong>`, sub:`${fmtMoney(p.monto)} · vencido ${toHoy(p.fecha)}` });
  });
  S.pagos.filter(p=>p.estado==='pendiente'&&p.fecha<=man).forEach(p => {
    const al = S.alumnos.find(a=>a.id===p.alumnoId);
    if (al) notifs.push({ t:'a', msg:`Pago próximo — <strong>${al.nombre}</strong>`, sub:`${fmtMoney(p.monto)} · vence ${toHoy(p.fecha)}` });
  });
  S.clases.filter(c=>c.fecha===hoy).forEach(cl => {
    notifs.push({ t:'g', msg:`Clase hoy — <strong>${cl.titulo}</strong>`, sub:`${cl.hora} · ${cl.cancha} · ${cl.dur}min` });
  });
 
  const nRows = notifs.length ? notifs.map(n => `
    <div class="row" style="cursor:default;align-items:flex-start;gap:14px">
      <div class="ndot ${n.t}" style="margin-top:6px"></div>
      <div class="row-info"><div class="row-name" style="font-size:14px">${n.msg}</div><div class="row-sub">${n.sub}</div></div>
    </div>`).join('') : '<p class="empty">Sin alertas pendientes 🎉</p>';
 
  document.getElementById('sec-notifs').innerHTML = `
    <div class="t-sec">Alertas (${notifs.length})</div>
    <div class="card-flush">${nRows}</div>
    <div class="t-sec">Configurar recordatorios</div>
    <div class="card">
      <div class="inf"><span class="inf-lbl">Aviso de pago vencido</span><label class="tog"><input type="checkbox" ${S.config.aviso_pago?'checked':''} onchange="toggleCfg('aviso_pago',this.checked)"><span class="tog-sl"></span></label></div>
      <div class="inf"><span class="inf-lbl">Recordatorio de clase</span><label class="tog"><input type="checkbox" ${S.config.aviso_clase?'checked':''} onchange="toggleCfg('aviso_clase',this.checked)"><span class="tog-sl"></span></label></div>
      <div class="inf"><span class="inf-lbl">Resumen semanal</span><label class="tog"><input type="checkbox" ${S.config.resumen?'checked':''} onchange="toggleCfg('resumen',this.checked)"><span class="tog-sl"></span></label></div>
    </div>
  `;
}
async function toggleCfg(k, v) {
  S.config[k]=v; await DB.put('config', S.config); toast('Guardado');
}

/* ═══════════════════════════════════════════════════════════════════
   FORMS — PROGRESO
═══════════════════════════════════════════════════════════════════ */
function formProgreso(alumnoId) {
  const prev  = (S.notas.filter(n=>n.alumnoId===alumnoId).sort((a,b)=>b.fecha.localeCompare(a.fecha))[0])||null;
  const habs  = ['Saque','Volea','Remate','Defensa','Posición'];
  const sliders = habs.map(h => {

    return `<label>${h} <span id="sv-${h}">${v}%</span></label>
      <input type="range" min="0" max="100" step="5" value="${v}" oninput="document.getElementById('sv-${h}').textContent=this.value+'%'" id="sl-${h}">`;
  }).join('');
  openModal('Actualizar progreso', `
    ${sliders}
    <label>Comentario</label>
    <textarea id="f-com" placeholder="Observaciones, ejercicios trabajados..."></textarea>
  `, async () => {
    const hd = {}; habs.forEach(h => hd[h] = parseInt(document.getElementById('sl-'+h).value));
    await DB.add('notas', { alumnoId, fecha:fmt(new Date()), habs:hd, comentario:$v('f-com').trim() });
    await loadAll(); closeModal(); toast('Progreso guardado');

  });
}

   DELETE
═══════════════════════════════════════════════════════════════════ */
async function eliminarAlumno(id) {
  const ok = await confirm('¿Eliminar este alumno? Se borrarán también sus pagos y notas.');
  if (!ok) return;
  await DB.del('alumnos', id);
  for (const p of S.pagos.filter(p=>p.alumnoId===id)) await DB.del('pagos',p.id);
  for (const n of S.notas.filter(n=>n.alumnoId===id)) await DB.del('notas',n.id);
  await loadAll(); toast('Alumno eliminado'); rAlumnos();
}
async function eliminarClase(id) {
  closeModal();
  const ok = await confirm('¿Eliminar esta clase?');
  if (!ok) return;
  const cl = await DB.get('clases',id);

   SERVICE WORKER (inline)
═══════════════════════════════════════════════════════════════════ */
function registerSW() {
  if (!('serviceWorker' in navigator)) return;
  const swCode = `
    const C='pc-v1',A=['./'];
    self.addEventListener('install',e=>{e.waitUntil(caches.open(C).then(c=>c.addAll(A)));self.skipWaiting()});
    self.addEventListener('activate',e=>{e.waitUntil(caches.keys().then(ks=>Promise.all(ks.filter(k=>k!==C).map(k=>caches.delete(k)))));self.clients.claim()});
    self.addEventListener('fetch',e=>{e.respondWith(caches.match(e.request).then(r=>r||fetch(e.request).catch(()=>caches.match('./'))))});
  `;
  const blob = new Blob([swCode], { type:'application/javascript' });
  const url  = URL.createObjectURL(blob);
  navigator.serviceWorker.register(url).catch(()=>{});
}

/* ═══════════════════════════════════════════════════════════════════
   GOOGLE CALENDAR — OAuth2 + API REAL
   Usa un calendario DEDICADO "PadelCoach" — nunca toca el primario
═══════════════════════════════════════════════════════════════════ */
const GCAL = (() => {
  const CLIENT_ID      = '456704676078-kh72v2ar7jmjtpq9dj2pid5a77cr8obc.apps.googleusercontent.com';
  const SCOPES         = 'https://www.googleapis.com/auth/calendar https://www.googleapis.com/auth/userinfo.email https://www.googleapis.com/auth/userinfo.profile';
  const CAL_NAME       = 'PadelCoach 🎾';
  const GCAL_EVENT_TAG = 'padelcoach';
 
  let tokenClient  = null;
  let accessToken  = null;
  let userInfo     = null;
  let dedicatedCalId = null; // ID del calendario "PadelCoach"
  let gapiReady    = false;
 
  // ── Init GIS ─────────────────────────────────────────────────────
  function initGIS() {
    return new Promise(res => {
      if (window.google?.accounts) { buildClient(); res(); return; }
      const s = document.createElement('script');
      s.src = 'https://accounts.google.com/gsi/client';
      s.onload = () => { buildClient(); res(); };
      s.onerror = () => res();
      document.head.appendChild(s);
    });
  }
 
  function buildClient() {
    if (!window.google?.accounts?.oauth2) return;
    tokenClient = google.accounts.oauth2.initTokenClient({
      client_id: CLIENT_ID,
      scope: SCOPES,
      callback: handleToken
    });
    gapiReady = true;
  }
 
  function handleToken(resp) {
    if (resp.error) { toast('Error al conectar con Google', 'err'); renderPanel(); return; }
    accessToken = resp.access_token;
    const exp = Date.now() + (resp.expires_in - 60) * 1000;
    localStorage.setItem('gcal_token', JSON.stringify({ token: accessToken, exp }));
    Promise.all([fetchUserInfo(), ensureDedicatedCalendar()])
      .then(() => { renderPanel(); GCAL.importFromGCal().catch(()=>{}); });
  }
 
  function loadToken() {
    try {
      const saved = JSON.parse(localStorage.getItem('gcal_token') || 'null');
      if (saved && saved.exp > Date.now()) { accessToken = saved.token; return true; }
    } catch(e) {}
    accessToken = null;
    return false;
  }
 
  async function fetchUserInfo() {
    try {
      userInfo = await apiFetch('https://www.googleapis.com/oauth2/v2/userinfo');
      localStorage.setItem('gcal_user', JSON.stringify(userInfo));
    } catch(e) {}
  }
 
  function loadUserInfo() {
    try { userInfo = JSON.parse(localStorage.getItem('gcal_user') || 'null'); } catch(e) {}
  }
 
  function loadCalId() {
    dedicatedCalId = localStorage.getItem('gcal_cal_id') || null;
  }
 
  // ── Garantizar token válido antes de cada operación ─────────────
  function ensureToken() {
    return new Promise((resolve, reject) => {
      const saved = JSON.parse(localStorage.getItem('gcal_token') || 'null');
      if (saved && saved.exp > Date.now()) {
        accessToken = saved.token;
        resolve();
        return;
      }
      // Token expirado — renovar silenciosamente
      if (!tokenClient) { reject(new Error('no token client')); return; }
      const origCallback = tokenClient.callback;
      tokenClient.callback = (resp) => {
        tokenClient.callback = origCallback;
        if (resp.error) { reject(new Error(resp.error)); return; }
        accessToken = resp.access_token;
        const exp = Date.now() + (resp.expires_in - 60) * 1000;
        localStorage.setItem('gcal_token', JSON.stringify({ token: accessToken, exp }));
        resolve();
      };
      tokenClient.requestAccessToken({ prompt: '' });
    });
  }
 
  // ── API helper ───────────────────────────────────────────────────
  async function apiFetch(url, opts = {}) {
    await ensureToken();
    const res = await fetch(url, {
      ...opts,
      headers: {
        'Authorization': `Bearer ${accessToken}`,
        'Content-Type': 'application/json',
        ...(opts.headers || {})
      }
    });
    if (res.status === 401) {
      accessToken = null;
      localStorage.removeItem('gcal_token');
      // Intentar renovar token silenciosamente
      if (tokenClient) {
        tokenClient.requestAccessToken({ prompt: '' });
      }
      throw new Error('401');
    }
    if (!res.ok) throw new Error(res.status);
    if (res.status === 204) return null;
    return res.json();
  }
 
  // ── Buscar o crear calendario dedicado "PadelCoach 🎾" ───────────
  // NUNCA modifica el calendario primario ni ningún otro.
  async function ensureDedicatedCalendar() {
    // 1. Buscar si ya existe entre los calendarios del usuario
    const list = await apiFetch('https://www.googleapis.com/calendar/v3/users/me/calendarList');
    const existing = (list.items || []).find(c => c.summary === CAL_NAME);
    if (existing) {
      dedicatedCalId = existing.id;
      localStorage.setItem('gcal_cal_id', dedicatedCalId);
      return;
    }
    // 2. No existe → crearlo. Esto crea un calendario nuevo, separado del primario.
    const created = await apiFetch('https://www.googleapis.com/calendar/v3/calendars', {
      method: 'POST',
      body: JSON.stringify({
        summary: CAL_NAME,
        description: 'Clases de pádel gestionadas por PadelCoach App. No modificar manualmente.',
        timeZone: Intl.DateTimeFormat().resolvedOptions().timeZone
      })
    });
    dedicatedCalId = created.id;
    localStorage.setItem('gcal_cal_id', dedicatedCalId);
  }
 
  // ── Construir evento Google Calendar ─────────────────────────────
  function buildEvent(cl) {
    const al = cl.alumnoId ? S.alumnos.find(a => a.id === cl.alumnoId) : null;
    const startDt = new Date(`${cl.fecha}T${cl.hora}:00`);
    const endDt   = new Date(startDt.getTime() + cl.dur * 60000);
    const pad = n => String(n).padStart(2, '0');
    const fmtLocal = d =>
      `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}T${pad(d.getHours())}:${pad(d.getMinutes())}:00`;
    return {
      summary: `🎾 ${cl.titulo}`,
      location: cl.cancha || '',
      description: [
        `Tipo: ${cl.tipo || 'clase'}`,
        al ? `Alumno: ${al.nombre}` : '',
        cl.notas ? `Notas: ${cl.notas}` : '',
        `[${GCAL_EVENT_TAG}:${cl.id}]`   // tag interno para identificar el evento
      ].filter(Boolean).join('\n'),
      start: { dateTime: fmtLocal(startDt), timeZone: Intl.DateTimeFormat().resolvedOptions().timeZone },
      end:   { dateTime: fmtLocal(endDt),   timeZone: Intl.DateTimeFormat().resolvedOptions().timeZone },
      colorId: '2'  // verde sage
    };
  }
 
  // ── Obtener eventos existentes del calendario dedicado ───────────
  // Opera SOLO sobre dedicatedCalId, nunca sobre 'primary'
  async function fetchExistingEvents() {
    if (!dedicatedCalId) await ensureDedicatedCalendar();
    const now = new Date();
    const min = new Date(now.getFullYear() - 1, 0, 1).toISOString();
    const max = new Date(now.getFullYear() + 1, 11, 31).toISOString();
    const params = new URLSearchParams({
      timeMin: min, timeMax: max,
      singleEvents: true, maxResults: 2500
    });
    const data = await apiFetch(
      `https://www.googleapis.com/calendar/v3/calendars/${encodeURIComponent(dedicatedCalId)}/events?${params}`
    );
    return data.items || [];
  }
 
  // ── Sync: crear/actualizar clases en el calendario dedicado ──────
  async function syncAll() {
    setLoading(true);
    try {
      if (!dedicatedCalId) await ensureDedicatedCalendar();
      const existing = await fetchExistingEvents();
 
      // Mapear eventos existentes por tag [padelcoach:ID]
      const gcalMap = {};
      existing.forEach(ev => {
        const m = (ev.description || '').match(/\[padelcoach:(\d+)\]/);
        if (m) gcalMap[m[1]] = ev.id;
      });
 
      let created = 0, updated = 0;
      const calEnc = encodeURIComponent(dedicatedCalId);
 
      for (const cl of S.clases) {
        const event = buildEvent(cl);
        const existingGcalId = gcalMap[String(cl.id)];
        if (existingGcalId) {
          // Actualizar evento existente — solo dentro del calendario dedicado
          await apiFetch(
            `https://www.googleapis.com/calendar/v3/calendars/${calEnc}/events/${existingGcalId}`,
            { method: 'PUT', body: JSON.stringify(event) }
          );
          updated++;
        } else {
          // Crear evento nuevo — solo dentro del calendario dedicado
          await apiFetch(
            `https://www.googleapis.com/calendar/v3/calendars/${calEnc}/events`,
            { method: 'POST', body: JSON.stringify(event) }
          );
          created++;
        }
      }
 
      toast(`✓ ${created} creados · ${updated} actualizados en "${CAL_NAME}"`, 'ok');
      renderPanel(true);
    } catch(e) {
      console.error(e);
      toast('Error de sincronización', 'err');
      setLoading(false);
    }
  }
 
  // ── Sync clase individual ────────────────────────────────────────
  async function syncOne(cl) {
    if (!accessToken) { toast('Conectá Google Calendar primero', 'err'); return; }
    try {
      if (!dedicatedCalId) await ensureDedicatedCalendar();
      const existing = await fetchExistingEvents();
      const found = existing.find(ev => (ev.description||'').includes(`[${GCAL_EVENT_TAG}:${cl.id}]`));
      const event = buildEvent(cl);
      const calEnc = encodeURIComponent(dedicatedCalId);
      if (found) {
        await apiFetch(`https://www.googleapis.com/calendar/v3/calendars/${calEnc}/events/${found.id}`,
          { method: 'PUT', body: JSON.stringify(event) });
      } else {
        await apiFetch(`https://www.googleapis.com/calendar/v3/calendars/${calEnc}/events`,
          { method: 'POST', body: JSON.stringify(event) });
      }
      toast('Clase sincronizada ✓');
    } catch(e) {

  // ── Importar eventos de GCal → app ──────────────────────────────
  async function importFromGCal() {
    if (!accessToken) { toast('Conectá Google Calendar primero', 'err'); return; }
    setImporting(true);
    try {
      if (!dedicatedCalId) await ensureDedicatedCalendar();
      const events = await fetchExistingEvents();
 
      // ── 1. Detectar clases eliminadas en GCal → borrar en app ────
      // Para cada clase de la app que tenga tag en GCal, verificar si sigue existiendo
      const gcalIds = new Set(events.map(ev => {
        const m = (ev.description || '').match(/\[padelcoach:(\d+)\]/);
        return m ? m[1] : null;
      }).filter(Boolean));
 
      let deleted = 0;
      for (const cl of S.clases) {
        // Solo verificar clases que fueron sincronizadas con GCal (tienen tag)
        // Si la clase tiene gcal_imported es una clase importada desde GCal, no tiene tag propio
        if (!cl.gcal_imported) {
          // Clase creada en la app → verificar si su evento sigue en GCal
          if (!gcalIds.has(String(cl.id))) {
            // El evento fue eliminado en GCal → borrar en la app también
            await DB.del('clases', cl.id);
            deleted++;
          }
        }
      }
      if (deleted > 0) {
        await loadAll();
        toast(`🗑️ ${deleted} clase${deleted>1?'s':''} eliminada${deleted>1?'s':''} desde Google Calendar`);
      }
 
      // ── 2. Importar eventos nuevos de GCal → app ─────────────────
      const externos = events.filter(ev => {
        const desc = ev.description || '';
        return !desc.includes(`[${GCAL_EVENT_TAG}:`);
      });
 
      let imported = 0;
      for (const ev of externos) {
        if (!ev.start?.dateTime && !ev.start?.date) continue;
 
        const startDt = new Date(ev.start.dateTime || ev.start.date);
        const endDt   = new Date(ev.end?.dateTime || ev.end?.date || startDt);
        const dur     = Math.round((endDt - startDt) / 60000) || 60;
        const pad     = n => String(n).padStart(2,'0');
        const fecha   = `${startDt.getFullYear()}-${pad(startDt.getMonth()+1)}-${pad(startDt.getDate())}`;
        const hora    = `${pad(startDt.getHours())}:${pad(startDt.getMinutes())}`;
        const titulo  = (ev.summary || 'Clase importada').replace(/^🎾\s*/, '');
 
        // Verificar si ya existe una clase en la app con el mismo gcal id o misma fecha/hora
        const yaExiste = S.clases.some(c => c.gcal_imported === ev.id || (c.fecha === fecha && c.hora === hora));
        if (yaExiste) continue;
 
        const nuevaClase = {
          titulo,
          fecha,
          hora,
          dur,
          cancha: ev.location || 'Cancha 1',
          tipo: 'clase',
          alumnoId: null,
          estado: 'confirmada',
          notas: (ev.description || '').replace(/\n?\[.*?\]/g, '').trim(),
          gcal_imported: ev.id
        };
 
        await DB.add('clases', nuevaClase);
        imported++;
      }
 
      if (imported > 0) {
        await loadAll();
        toast(`✓ ${imported} clase${imported>1?'s':''} importada${imported>1?'s':''} desde Google Calendar`, 'ok');
        renderPanel();
      } else {
        toast('No hay clases nuevas para importar', 'ok');
      }
    } catch(e) {
      console.error(e);
      toast('Error al importar desde Google Calendar', 'err');
    } finally {
      setImporting(false);
    }
  }

      if (metodo === 'efectivo') btns[0].classList.add('selected-efectivo');
      if (metodo === 'transferencia') btns[1].classList.add('selected-transferencia');
      if (metodo === 'nopago') btns[2].classList.add('selected-nopago');
    },
    async confirm() {
      const hoy = fmt(new Date());
      let guardados = 0;
      for (const [key, metodo] of Object.entries(selections)) {
        if (!metodo) continue;
        const alumnoId = key === 'grupal' ? null : parseInt(key);
        if (!alumnoId) continue;
        if (metodo === 'nopago') {
          registrarNoPago(alumnoId, currentClase.id, currentClase.titulo, currentClase.fecha);
        } else {
          const metodoMap = { efectivo:'Efectivo', transferencia:'Transferencia', mercadopago:'MercadoPago', paquete:'Paquete', cuota:'Cuota mensual' };
          await DB.add('pagos', {
            alumnoId,
            monto: 0,
            fecha: hoy,
            metodo: metodoMap[metodo] || metodo,
            estado: 'pagado'
          });
          guardados++;
        }
      }
      await loadAll();
      overlay.classList.remove('on');
      if (currentClase) markShown(currentClase.id + '_' + currentClase.fecha);
      toast(guardados > 0 ? `✓ ${guardados} pago${guardados>1?'s':''} registrado${guardados>1?'s':''}` : 'Sin pagos registrados');
    },
    skip() {
      overlay.classList.remove('on');
      if (currentClase) markShown(currentClase.id + '_' + currentClase.fecha);
    }
  };
 
  // Revisar cada minuto si alguna clase acaba de terminar
  function checkClasesFin() {
    const now = new Date();
    const hoy = fmt(now);
    const shown = getShown();
    const clasesHoy = S.clases.filter(c => c.fecha === hoy && c.estado !== 'cancelada');
 
    for (const cl of clasesHoy) {
      const key = cl.id + '_' + cl.fecha;
      if (shown[key]) continue; // ya se mostró hoy
 
      const [h, m] = cl.hora.split(':').map(Number);
      const inicio = new Date(now); inicio.setHours(h, m, 0, 0);
      const fin = new Date(inicio.getTime() + cl.dur * 60000);
      const diffMs = now - fin;
 
      // Mostrar si la clase terminó hace entre 0 y 10 minutos
      if (diffMs >= 0 && diffMs <= 10 * 60 * 1000) {
        showPopup(cl);
        break; // mostrar de a uno
      }
    }
  }
 
  // Arrancar el chequeo cada 60 segundos
  document.addEventListener('DOMContentLoaded', () => {
    setTimeout(() => {
      checkClasesFin();
      setInterval(checkClasesFin, 60 * 1000);
    }, 3000); // esperar 3s al arrancar para que carguen los datos
  });
})();
